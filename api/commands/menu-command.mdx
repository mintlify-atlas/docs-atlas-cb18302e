## Overview

`ContextMenuCommand` is the base class for creating context menu commands in Discord. These are commands that appear in the right-click menu on users or messages.

## Class Signature

```typescript
export abstract class ContextMenuCommand {
  middlewares: (keyof RegisteredMiddlewares)[] = [];
  name!: string;
  type!: ApplicationCommandType.User | ApplicationCommandType.Message;
  nsfw?: boolean;
  integrationTypes: ApplicationIntegrationType[] = [];
  contexts: InteractionContextType[] = [];
  description!: string;
  defaultMemberPermissions?: bigint;
  botPermissions?: bigint;
  dm?: boolean;
  name_localizations?: Partial<Record<LocaleString, string>>;
  description_localizations?: Partial<Record<LocaleString, string>>;
  props: ExtraProps = {};
  
  abstract run?(context: MenuCommandContext<any>): any;
}
```

## Examples

### User Context Menu Command

```typescript
import { ContextMenuCommand, Declare, type MenuCommandContext } from 'seyfert';
import { ApplicationCommandType } from 'seyfert/lib/types';

@Declare({
  name: 'User Info',
  type: ApplicationCommandType.User,
})
export default class UserInfoCommand extends ContextMenuCommand {
  async run(ctx: MenuCommandContext<UserCommandInteraction>) {
    // ctx.target is typed as UserStructure
    const user = ctx.target;
    
    await ctx.write({
      embeds: [{
        title: 'User Information',
        fields: [
          { name: 'Username', value: user.username },
          { name: 'ID', value: user.id },
          { name: 'Created', value: user.createdAt.toDateString() },
        ],
      }],
      flags: MessageFlags.Ephemeral,
    });
  }
}
```

### Message Context Menu Command

```typescript
import { ContextMenuCommand, Declare, type MenuCommandContext } from 'seyfert';
import { ApplicationCommandType, MessageCommandInteraction } from 'seyfert/lib/types';

@Declare({
  name: 'Copy Message',
  type: ApplicationCommandType.Message,
})
export default class CopyMessageCommand extends ContextMenuCommand {
  async run(ctx: MenuCommandContext<MessageCommandInteraction>) {
    // ctx.target is typed as MessageStructure
    const message = ctx.target;
    
    if (!message.content) {
      return ctx.write({
        content: 'This message has no text content.',
        flags: MessageFlags.Ephemeral,
      });
    }
    
    await ctx.write({
      content: `**Message from ${message.author.username}:**\n${message.content}`,
      flags: MessageFlags.Ephemeral,
    });
  }
}
```

### With Modal Response

```typescript
import { ContextMenuCommand, Declare, type MenuCommandContext } from 'seyfert';
import { ApplicationCommandType, MessageCommandInteraction } from 'seyfert/lib/types';
import { ActionRow, TextInput, TextInputStyle } from 'seyfert/lib/builders';

@Declare({
  name: 'Report Message',
  type: ApplicationCommandType.Message,
})
export default class ReportMessageCommand extends ContextMenuCommand {
  async run(ctx: MenuCommandContext<MessageCommandInteraction>) {
    const message = ctx.target;
    
    const modal = await ctx.modal({
      title: 'Report Message',
      customId: 'report-modal',
      components: [
        new ActionRow().setComponents([
          new TextInput()
            .setCustomId('reason')
            .setLabel('Reason for report')
            .setStyle(TextInputStyle.Paragraph)
            .setRequired(true),
        ]),
      ],
    }, { time: 5 * 60 * 1000 });
    
    if (modal) {
      const reason = modal.fields.getTextInputValue('reason');
      // Process report...
      
      await modal.write({
        content: 'Thank you for your report.',
        flags: MessageFlags.Ephemeral,
      });
    }
  }
}
```
