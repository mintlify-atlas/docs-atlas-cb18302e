## Overview

`CommandContext` is the context object passed to command handlers when a slash command is invoked. It provides access to the interaction data, parsed options, response methods, and various helper utilities.

## Class Signature

```typescript
export class CommandContext<
  T extends OptionsRecord = {},
  M extends keyof RegisteredMiddlewares = never,
> extends BaseContext
```

## Type Parameters

## Response Methods

### write

```typescript
async write<WR extends boolean = false>(
  body: InteractionCreateBodyRequest,
  withResponse?: WR,
): Promise<When<WR, WebhookMessageStructure | MessageStructure, void>>
```

Sends an initial response to the interaction.

<ParamField path="body" type="InteractionCreateBodyRequest" required>
  The message payload to send.
</ParamField>

<ParamField path="withResponse" type="boolean" default="false">
  If `true`, returns the created message object.
</ParamField>

**Example:**
```typescript
await ctx.write({
  content: 'Hello, World!',
  embeds: [embed],
});
```

### deferReply

```typescript
async deferReply<WR extends boolean = false>(
  ephemeral?: boolean,
  withResponse?: WR,
): Promise<When<WR, WebhookMessageStructure | MessageStructure, void>>
```

Defers the interaction response, showing a loading state. Must edit the response later.

<ParamField path="ephemeral" type="boolean" default="false">
  If `true`, the response will be ephemeral (only visible to the command user).
</ParamField>

<ParamField path="withResponse" type="boolean" default="false">
  If `true`, returns the deferred message object.
</ParamField>

**Example:**
```typescript
await ctx.deferReply();
// ... perform long operation
await ctx.editResponse({ content: 'Done!' });
```

## Type Guards

### isChat

```typescript
isChat(): this is CommandContext<T, M>
```

Always returns `true` for CommandContext. Used for type narrowing.

### inGuild

```typescript
inGuild(): this is GuildCommandContext<T, M>
```

Returns `true` if the command was invoked in a guild (not DM). Narrows type to `GuildCommandContext` which guarantees `guildId` and `member` are defined.

**Example:**
```typescript
if (ctx.inGuild()) {
  // TypeScript knows guildId and member are not undefined here
  const guild = await ctx.guild(); // guaranteed to return GuildStructure
  console.log(ctx.member.roles); // member is guaranteed to exist
}
```

## GuildCommandContext

When `inGuild()` returns true, the context is narrowed to `GuildCommandContext`:

```typescript
export interface GuildCommandContext<
  T extends OptionsRecord = {},
  M extends keyof RegisteredMiddlewares = never,
> extends Omit<MakeRequired<CommandContext<T, M>, 'guildId' | 'member'>, 'guild' | 'me'> {
  guild(mode?: 'rest' | 'flow'): Promise<GuildStructure<'cached' | 'api'>>;
  guild(mode: 'cache'): ReturnCache<GuildStructure<'cached'> | undefined>;
  
  me(mode?: 'rest' | 'flow'): Promise<GuildMemberStructure>;
  me(mode: 'cache'): ReturnCache<GuildMemberStructure | undefined>;
}
```

This provides stronger type guarantees for guild-only commands.

## Complete Example

```typescript
import { Command, Declare, Options, type CommandContext } from 'seyfert';
import { ApplicationCommandOptionType } from 'seyfert/lib/types';

const options = {
  message: {
    type: ApplicationCommandOptionType.String,
    description: 'The message to echo',
    required: true,
  },
  ephemeral: {
    type: ApplicationCommandOptionType.Boolean,
    description: 'Send as ephemeral message',
    required: false,
  },
};

@Declare({
  name: 'echo',
  description: 'Echoes your message',
})
@Options(options)
export default class EchoCommand extends Command {
  async run(ctx: CommandContext<typeof options>) {
    // Access typed options
    const message = ctx.options.message;
    const ephemeral = ctx.options.ephemeral ?? false;
    
    // Check if in guild
    if (ctx.inGuild()) {
      const guild = await ctx.guild();
      console.log(`Command used in ${guild.name}`);
    }
    
    // Defer if needed
    if (message.length > 1000) {
      await ctx.deferReply(ephemeral);
      // ... process long message
      await ctx.editResponse({ content: message });
    } else {
      await ctx.write({
        content: message,
        flags: ephemeral ? MessageFlags.Ephemeral : undefined,
      });
    }
    
    // Send a followup
    await ctx.followup({
      content: 'Message sent!',
      ephemeral: true,
    });
  }
}
```
