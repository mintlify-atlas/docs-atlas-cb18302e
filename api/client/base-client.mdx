## BaseClient

The `BaseClient` class is the foundation for all Seyfert client types. It provides core functionality for handling commands, components, caching, REST API calls, and more.

### Constructor

```typescript
new BaseClient(options?: BaseClientOptions)
```

</ParamField>

### Methods

#### start()

Starts the client by loading languages, commands, and components.

```typescript
await client.start(options?: {
  langsDir?: string;
  commandsDir?: string;
  componentsDir?: string;
  token?: string;
  connection?: any;
}): Promise<void>
```

<ParamField path="options.langsDir" type="string">
  Directory containing language files
</ParamField>

<ParamField path="options.commandsDir" type="string">
  Directory containing command files
</ParamField>

<ParamField path="options.componentsDir" type="string">
  Directory containing component files
</ParamField>

<ParamField path="options.token" type="string">
  Discord bot token
</ParamField>

#### setServices()

Configures services like REST API, cache, and middleware.

```typescript
client.setServices(options: ServicesOptions): void
```

<ParamField path="options.rest" type="ApiHandler">
  Custom REST API handler
</ParamField>

<ParamField path="options.cache" type="object">
  Cache configuration

  - `disabledCache`: Disable specific cache types
</ParamField>

<ParamField path="options.langs" type="object">
  Language configuration
  - `default`: Default language

<ParamField path="options.middlewares" type="Record<string, MiddlewareContext>">

<ParamField path="options.handleCommand" type="typeof HandleCommand">

## BaseClientOptions

Configuration options for BaseClient.

```typescript
interface BaseClientOptions {
  context?: (interaction: Interaction | Message) => Record<string, unknown>;
  globalMiddlewares?: readonly string[];
  commands?: {
    defaults?: {
      onBeforeMiddlewares?: (context: CommandContext | MenuCommandContext) => unknown;
      onBeforeOptions?: (context: CommandContext) => unknown;
      onRunError?: (context: CommandContext, error: unknown) => unknown;
      onPermissionsFail?: (context: CommandContext, permissions: PermissionStrings) => unknown;
      onBotPermissionsFail?: (context: CommandContext, permissions: PermissionStrings) => unknown;
      onInternalError?: (client: UsingClient, command: Command, error?: unknown) => unknown;
      onMiddlewaresError?: (context: CommandContext, error: string) => unknown;
      onOptionsError?: (context: CommandContext, metadata: any) => unknown;
      onAfterRun?: (context: CommandContext, error: unknown) => unknown;
      props?: ExtraProps;
    };
  };
  components?: {
    defaults?: {
      onBeforeMiddlewares?: (context: ComponentContext) => unknown;
      onRunError?: (context: ComponentContext, error: unknown) => unknown;
      onInternalError?: (client: UsingClient, error: unknown) => unknown;
      onMiddlewaresError?: (context: ComponentContext, error: string) => unknown;
      onAfterRun?: (context: ComponentContext, error: unknown) => unknown;
    };
  };
  modals?: {
    defaults?: {
      onBeforeMiddlewares?: (context: ModalContext) => unknown;
      onRunError?: (context: ModalContext, error: unknown) => unknown;
      onInternalError?: (client: UsingClient, error: unknown) => unknown;
      onMiddlewaresError?: (context: ModalContext, error: string) => unknown;
      onAfterRun?: (context: ModalContext, error: unknown) => unknown;
    };
  };
  allowedMentions?: AllowedMentions;
  getRC?(): Awaitable<InternalRuntimeConfig | InternalRuntimeConfigHTTP>;
}
```

<ParamField path="context" type="function">
  Function to provide custom context for interactions
</ParamField>

<ParamField path="globalMiddlewares" type="readonly string[]">
  Array of middleware names to apply globally
</ParamField>

<ParamField path="commands.defaults" type="object">
  Default handlers for command lifecycle events
</ParamField>

<ParamField path="components.defaults" type="object">
  Default handlers for component lifecycle events
</ParamField>

<ParamField path="modals.defaults" type="object">
  Default handlers for modal lifecycle events
</ParamField>

<ParamField path="allowedMentions" type="AllowedMentions">
  Default allowed mentions configuration
</ParamField>

<ParamField path="getRC" type="function">
  Custom function to retrieve runtime configuration
</ParamField>

## Example

```typescript
import { BaseClient } from 'seyfert';

const client = new BaseClient({
  commands: {
    defaults: {
      onRunError: (context, error) => {
        console.error('Command error:', error);
      },
    },
  },
});

await client.start({
  token: process.env.BOT_TOKEN,
});

await client.uploadCommands();
```
