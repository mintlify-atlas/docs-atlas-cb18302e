<div className="mt-16 mb-16 max-w-5xl mx-auto px-6">
  <h2 className="text-3xl font-bold text-gray-900 dark:text-white mb-4">Quick Start</h2>
  <p className="text-lg text-gray-600 dark:text-gray-400 mb-8">Get your Discord bot running in minutes</p>
  
  <Steps>
    <Step title="Install Seyfert">
      Choose your favorite package manager to install Seyfert:
      
      <CodeGroup>
      ```bash npm
      npm install seyfert
      ```
      
      ```bash pnpm
      pnpm add seyfert
      ```
      
      ```bash yarn
      yarn add seyfert
      ```
      
      ```bash bun
      bun add seyfert
      ```
      
      ```bash deno
      deno add npm:seyfert
      ```
      </CodeGroup>
      
      <Note>
        Node.js v18 or higher is required. We recommend using the LTS version for the best experience.
      </Note>
    </Step>
    
    <Step title="Create your bot client">
      Create a new file and set up your Discord bot client:
      
      ```typescript index.ts
      import { Client, config } from 'seyfert';
      
      const client = new Client();
      
      client.start({
        connection: config.bot({
          token: process.env.DISCORD_TOKEN ?? '',
          intents: ['Guilds', 'GuildMessages', 'MessageContent']
        })
      });
      ```
      
      <Info>
        Make sure to set your Discord bot token in the `DISCORD_TOKEN` environment variable. You can get your token from the [Discord Developer Portal](https://discord.com/developers/applications).
      </Info>
    </Step>
    
    <Step title="Create your first command">
      Create a command file to respond to user interactions:
      
      ```typescript commands/ping.ts
      import { Command, type CommandContext } from 'seyfert';
      
      export default class PingCommand extends Command {
        name = 'ping';
        description = 'Check the bot latency';
        
        async run(ctx: CommandContext) {
          const latency = ctx.client.gateway.latency;
          await ctx.write({
            content: `🏓 Pong! Latency: ${latency}ms`
          });
        }
      }
      ```
      
      Commands are automatically loaded from your commands directory. Just create the file and start your bot!
    </Step>
    
    <Step title="Run your bot">
      Start your bot and test the ping command:
      
      ```bash
      node index.js
      ```
      
      Your bot should now be online! Try using `/ping` in your Discord server to see it respond with the latency.
      
      <Tip>
        Use a process manager like PM2 or systemd for production deployments to keep your bot running reliably.
      </Tip>
    </Step>
  </Steps>
</div>
